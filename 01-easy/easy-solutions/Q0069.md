#### Python3:

```
class Solution:
    def mySqrt(self, x: int) -> int:
        left, right = 1, x
      
        while left <= right:
            mid = left + (right - left) // 2
          
            if mid == x // mid: return mid
            elif mid > x // mid: right = mid - 1
            else: left = mid + 1
      
        return right
```

#### Java:

```
class Solution {
    public int mySqrt(int x) {
        int left = 1, right = x;
      
        while (left <= right) {
            int mid = left + (right - left) / 2;
          
            if (mid == x / mid) return mid;
            else if (mid > x / mid) right = mid - 1;
            else left = mid + 1;
        }
      
        return right;
    }
}
```
