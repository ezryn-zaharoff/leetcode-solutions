## Q0383: [Ransom Note](https://leetcode.com/problems/ransom-note/)

```
Difficulty Level: EASY
```

```
Description:

Given two strings ransomNote and magazine, return true if ransomNote can be constructed by using the letters
from magazine and false otherwise.

Each letter in magazine can only be used once in ransomNote.
```

#### Python3:

```
class Solution:
    def canConstruct(self, ransomNote: str, magazine: str) -> bool:
        for ch in ransomNote:
            if ch in magazine:
                magazine = magazine.replace(ch, '', 1)
            else:
                return False
                
        return True
```

#### Java:

```
class Solution {
    public boolean canConstruct(String ransomNote, String magazine) {
        for (char ch : ransomNote.toCharArray()) {
            String c = Character.toString(ch);
            if (magazine.contains(c)) {
                magazine = magazine.replaceFirst(c, "");
            } else {
                return false;
            }
        }

        return true;
    }
}
```

```
class Solution {
public:
    bool canConstruct(string ransomNote, string magazine) {
        for (auto& ch : ransomNote) {
            auto c = magazine.find(ch);
            if (c != string::npos) {
                magazine[c] = 1;
            } else {
                return false;
            }
        }

        return true;
    }
};
```

```
func canConstruct(ransomNote string, magazine string) bool {
    for _, ch := range ransomNote {
        c := string(ch)
        i := strings.Index(magazine, c)

        if i == -1 {return false}

        magazine = magazine[:i] + magazine[i+1:]
    }

    return true
}
```

```
public class Solution {
    public bool CanConstruct(string ransomNote, string magazine) {
        List<char> list = new List<char>(magazine);

        foreach (char ch in ransomNote) {
            if (list.Contains(ch)) {
                list.Remove(ch);
            } else {
                return false;
            }
        }

        return true;
    }
}
```

#### Swift:

```
class Solution {
    func canConstruct(_ ransomNote: String, _ magazine: String) -> Bool {
        var mag = magazine

        for ch in ransomNote {
            guard let i = mag.firstIndex(where: { $0 == ch })
            else {
                return false
            }

            mag.remove(at: i)
        }
        
        return true
    }
}
```
